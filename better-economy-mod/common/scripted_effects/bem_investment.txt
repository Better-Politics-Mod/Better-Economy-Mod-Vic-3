bem_private_investment_oa = {
	set_local_variable = {
		name = multi
		value = {
			value = bem_state_investment_levels
			subtract = 1
		}
	}
	set_local_variable = {
		name = mult
		value = {
			value = bem_state_investment_levels
		}
	}
	every_scope_building = {
		remove_modifier = bem_investment_level
	}
	every_scope_building = {
		limit = {
			OR = {
				has_active_production_method = pm_bem_industrial_investment
				has_active_production_method = pm_bem_resource_investment
				has_active_production_method = pm_bem_service_investment
			}
		}
		add_modifier = {
			name = bem_investment_level
			multiplier = local_var:multi
		}
	}
	if = {
		limit = {
			NOT = {
				has_building = building_bem_bank
			}
		}
		create_building = {
			building = building_bem_bank
			level = 1
		}
	}
	every_scope_building = {
		limit = {
			is_building_type = building_bem_bank
		}
		remove_modifier = bem_bank_modi
		add_modifier = {
			name = bem_bank_modi
			multiplier = local_var:multi
		}
	}
	remove_modifier = bem_investment_stock
	add_modifier = {
		name = bem_investment_stock
		multiplier = local_var:mult
	}
	owner = {
		add_investment_pool = prev.bem_investment_weekly_loss
	}
}
