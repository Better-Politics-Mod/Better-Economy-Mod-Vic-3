je_bem_borrowing = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_internal_affairs
	
	scripted_button = bem_nationalize

	on_monthly_pulse = {
		effect = {
			scope:journal_entry = {
				if = {
					limit = {
						NOT = {
							has_modifier = bem_capitalist_investment
						}
					}
					add_modifier = {
						name = bem_capitalist_investment
						multiplier = 5
					}
				}
				if = {
					limit = {
						NOT = {
							owner = { has_modifier = bem_can_nationalize }
						}
						NOT = {
							has_modifier = bem_cant_nationalize
						}
					}
					add_modifier = {
						name = bem_cant_nationalize
					}
				}
				else_if = {
					limit = {
						owner = { has_modifier = bem_can_nationalize }
					}
					remove_modifier = bem_cant_nationalize
				}
			}
		}
	}

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = bem_previous_roi
				}
				set_variable = {
					name = bem_roi_change
					value = {
						value = owner.bem_roi
						subtract = var:bem_previous_roi
						divide = owner.bem_roi
					}
				}
			}

			if = {
				limit ={
					NOT = {
						has_variable = bem_economy_heat
					}
				}
				set_variable = {
					name = bem_economy_heat
					value = 5
				}
			}

			if = {
				limit = {
					has_variable = bem_previous_roi
				}
				change_variable = {
					name = bem_economy_heat
					multiply = {
						value = 1
						add = {
							value = bem_roi_change
							multiply = -1
							multiply = bem_rchg_mult
						}
					}
				}
				clamp_variable = {
					name = bem_economy_heat
					min = 1
					max = 20
				}
			}
			set_variable = {
				name = bem_heatchg
				value = {
					value = bem_roi_change
						multiply = -1
						multiply = bem_rchg_mult
				}
			}
			set_variable = {
				name = bem_roichg
				value = bem_roi_change
			}

			set_variable = {
				name = bem_previous_roi
				value = owner.bem_roi
			}
			

		}
	}

	should_be_pinned_by_default = yes

	complete = {

	}

	immediate = {

	}

	on_complete = {

	}

	invalid = {

	}

	current_value = {
	}

	goal_add_value = {

	}

	is_shown_when_inactive = {
		always = yes
	}
 
	possible = {
		always = yes
	}

	progressbar = yes

	weight = 100
}
