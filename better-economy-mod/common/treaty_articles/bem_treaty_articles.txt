overlord_transit_rights = {
    kind = directed
    cost = 0

	icon = "gfx/interface/icons/diplomatic_treaties_articles_icons/transit_rights_treaties.dds"

	flags = {
		is_transit_rights
	}

	maintenance_paid_by = target_country
    usage_limit = once_per_side

    unlocked_by_technologies = {
        international_relations
    }

	visible = {}

	possible = {
		is_junior_in_customs_union = no
		scope:other_country ?= {
			is_junior_in_customs_union = no
		}
	}

	can_withdraw = {
		NOT = {
			OR = {
				scope:withdrawing_country = {
					has_subject_relation_with = scope:non_withdrawing_country
				}
				scope:withdrawing_country.overlord = scope:non_withdrawing_country.overlord
			}
		}
	}

    can_ratify = {always = no}

    on_entry_into_force = {}
	
	on_withdrawal = {}
	on_break = {
		scope:withdrawing_country = {
			change_relations = { country = scope:non_withdrawing_country value = -10 }
		}
	}

	ai = {

        treaty_categories = {
			trade
		}

		evaluation_chance = {
			value = 0

			if = {
				limit = { capital.world_market_access <= 0 }
				add = 1.0
			}
		}

        article_ai_usage = {
            request
        }

        inherent_accept_score = {
			value = 0

			if = {
				limit = {
					scope:source_country ?= this
				}

				# We are doing this to be able to reuse loc across treaties and diplomatic actions
				scope:source_country = {
					save_temporary_scope_as = target
				}
				scope:target_country = {
					save_temporary_scope_as = actor
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"

					if = {
						limit = {
							OR = {
								has_attitude = {
									who = scope:target_country
									attitude = antagonistic
								}
								has_attitude = {
									who = scope:target_country
									attitude = domineering
								}
								has_attitude = {
									who = scope:target_country
									attitude = belligerent
								}
							}
						}
						add = -200
					}
				}
			}
			else_if = {
				limit = {
					scope:target_country ?= this
					"ai_transit_rights_value(scope:source_country)" > 0
				}
				scope:source_country = {
					save_temporary_scope_as = actor
				}
				scope:target_country = {
					save_temporary_scope_as = target
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_TRANSIT_RIGHTS_VALUE"
					value = "ai_transit_rights_value(scope:source_country)"
					max = 50
				}
			}
            else_if = {
                limit = { NOT = { exists = scope:is_treaty_active } }
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_DOES_NOT_WANT_ARTICLE"
					value = -300
				}
			}
		}
	}
}

overlord_trade_privilege = {
    kind = directed
    cost = 0

    relations_progress_per_day = 0
    relations_improvement_max = 0

	icon = "gfx/interface/icons/diplomatic_treaties_articles_icons/trade_agreement_treaties.dds"

	flags = {
		is_trade_privilege
	}

	maintenance_paid_by = target_country
	usage_limit = once_per_side

    unlocked_by_technologies = {
        international_relations
    }

	visible = {
		is_junior_in_customs_union = no
	}

	possible = {
		NOT = { has_law = law_type:law_isolationism }
		scope:other_country = {
			NOT = { has_law = law_type:law_isolationism }
		}
	}

    non_fulfillment = {
        consequences = freeze

        conditions = {
            weekly = {
				scope:article = {
					source_country = { has_law = law_type:law_isolationism }
				}
            }
        }
    }

	can_withdraw = {
		NOT = {
			scope:withdrawing_country = {
				has_subject_relation_with = scope:non_withdrawing_country
			}
		}
	}

    can_ratify = {always = no}

    on_entry_into_force = {
		add_lobby_appeasement_from_diplomacy_bidirectional = {
			FIRST = scope:article_options.source_country
			SECOND = scope:article_options.target_country
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_trade_agreement_formed
		}
	}

	on_withdrawal = {
		add_lobby_appeasement_from_diplomacy_bidirectional = {
			FIRST = scope:withdrawing_country
			SECOND = scope:non_withdrawing_country
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_trade_agreement_broken
		}
	}
	on_break = {
		scope:withdrawing_country = {
			change_relations = { country = scope:non_withdrawing_country value = -10 }
		}
	}

	ai = {

		treaty_categories = {
			trade
		}

		evaluation_chance = {
			value = 0.1
		}

		article_ai_usage = {
			request
		}

		inherent_accept_score = {
			value = 0

			if = {
				limit = {
					scope:source_country ?= this
				}

				# We are doing this to be able to reuse loc across treaties and diplomatic actions
				scope:source_country = {
					save_temporary_scope_as = target
				}
				scope:target_country = {
					save_temporary_scope_as = actor
				}

				if = {
					limit = {
						has_law_or_variant = law_type:law_isolationism
					}
					add = {
						desc = "law_isolationism"
						add = -100
					}
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_GDP"
					value = "scope:target_country.ai_gdp_comparison(root)"
					multiply = 2
					min = -10
					max = 10
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_MUTUAL_TRADE_VALUE"
					value = "mutual_trade_value_with_country(scope:target_country)"
					multiply = 50000
					divide = { # We modulate this value by GDP, clamped to a range to ensure it remains relevant for very small/very large economies
						value = gdp
						max = scope:target_country.gdp # We use whichever GDP is smaller
						min = 1000000
					}
					min = 1
					max = 10
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
					if = {
						limit = {
							OR = {
								has_attitude = {
									who = scope:target_country
									attitude = antagonistic
								}
								has_attitude = {
									who = scope:target_country
									attitude = domineering
								}
								has_attitude = {
									who = scope:target_country
									attitude = belligerent
								}
								has_attitude = {
									who = scope:target_country
									attitude = rebellious
								}
							}
						}
						value = -1000
					}
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_INCOMPATIBLE_ECONOMIC_SYSTEMS"
					if = {
						limit = {
							OR = {
								AND = {
									scope:target_country = { has_radical_leftist_economy = no }
									scope:target_country = { has_radical_leftist_economy = yes }
								}
								AND = {
									scope:target_country = { has_radical_leftist_economy = yes }
									scope:target_country = { has_radical_leftist_economy = no }
								}
							}
						}
						value = -50
					}
				}
			}
			else = {
				scope:source_country = {
					save_temporary_scope_as = actor
				}
				scope:target_country = {
					save_temporary_scope_as = target
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_GDP"
					value = "scope:source_country.ai_gdp_comparison(root)"
					multiply = 4
					min = -10
					max = 10
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_MUTUAL_TRADE_VALUE"
					value = "mutual_trade_value_with_country(scope:source_country)"
					multiply = 100000
					divide = { # We modulate this value by GDP, clamped to a range to ensure it remains relevant for very small/very large economies
						value = gdp
						max = scope:source_country.gdp # We use whichever GDP is smaller
						min = 1000000
					}
					min = 1
					max = 25
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
					if = {
						limit = {
							OR = {
								has_attitude = {
									who = scope:source_country
									attitude = antagonistic
								}
								has_attitude = {
									who = scope:source_country
									attitude = domineering
								}
								has_attitude = {
									who = scope:source_country
									attitude = belligerent
								}
								has_attitude = {
									who = scope:source_country
									attitude = rebellious
								}
							}
						}
						value = -1000
					}
				}

				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_INCOMPATIBLE_ECONOMIC_SYSTEMS"
					if = {
						limit = {
							OR = {
								AND = {
									scope:source_country = { has_radical_leftist_economy = no }
									scope:target_country = { has_radical_leftist_economy = yes }
								}
								AND = {
									scope:source_country = { has_radical_leftist_economy = yes }
									scope:target_country = { has_radical_leftist_economy = no }
								}
							}
						}
						value = -50
					}
				}
			}
		}
	}
}
